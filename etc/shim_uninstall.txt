# vim: set syntax=sh:

# uninstall shim
sudo apt purge shim shim-signed

# remove shimx64.efi
sudo rm /boot/efi/EFI/ubuntu/shimx64.efi

# reinstall grup
sudo grub-install /dev/sda

# check changes (should be \EFI\ubuntu\grubx64.efi)
efibootmgr -v

# remove older entries if needed
sudo efibootmgr -Bb FFFF

# remove delay
sudo efibootmgr -t 0

# update grub
sudo update-grub
